if (!globalAudioStream) { // 1. 取得原始麥克風，並強制關閉瀏覽器的「自動壓抑音量」機制 globalAudioStream = await navigator.mediaDevices.getUserMedia({ audio: { autoGainControl: false, // 關閉自動增益 (避免被瀏覽器壓低音量) noiseSuppression: false, // 關閉降噪 (保留最大真實音量) echoCancellation: false // 關閉回音消除 } }); // 2. 建立錄音專用的 Web Audio 引擎 const AudioContext = window.AudioContext || window.webkitAudioContext; const recCtx = new AudioContext(); // 3. 將原始聲音接入「擴大機」 const micSource = recCtx.createMediaStreamSource(globalAudioStream); const gainNode = recCtx.createGain(); // 🌟 核心關鍵：在這裡調整放大倍數！3.0 代表放大 3 倍 gainNode.gain.value = 3.0; // 4. 輸出放大後的聲音 const destNode = recCtx.createMediaStreamDestination(); micSource.connect(gainNode); gainNode.connect(destNode); // 5. 存入全域變數，保持通道暢通 globalAmplifiedStream = destNode.stream; } // 餵給錄音機的是「放大後的音軌」，而不是原始音軌 session.stream = globalAmplifiedStream; session.recorder = new MediaRecorder(session.stream);
