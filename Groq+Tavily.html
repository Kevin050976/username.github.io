async function askAI() { const groqKey = document.getElementById('groqKey').value.trim(); const tavilyKey = document.getElementById('tavilyKey').value.trim(); const userInputBox = document.getElementById('userInput'); const userInput = userInputBox.value.trim(); if (!groqKey || !userInput) return alert("請輸入 API Key 與問題"); saveKeys(); updateChatbox(`\n👤 **你:** ${userInput}\n`); userInputBox.value = ""; updateStatus("Groq 思考中..."); conversationHistory.push({ role: "user", content: userInput }); try { // --- 第一次請求：判斷是否需要搜尋 --- const response = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.1-8b-instant", // 改用 8B，TPM 額度較高 messages: conversationHistory, tools: [{ type: "function", function: { name: "web_search", description: "搜尋網路即時資訊", parameters: { type: "object", properties: { query: { type: "string" } }, required: ["query"] } } }] }) }); const data = await response.json(); const message = data.choices[0].message; if (message.tool_calls) { const toolCall = message.tool_calls[0]; const query = JSON.parse(toolCall.function.arguments).query; updateStatus(`🔍 搜尋中：${query}...`); conversationHistory.push(message); // --- 第二次請求：Tavily 搜尋 --- const searchRes = await fetch("https://api.tavily.com/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ api_key: tavilyKey, query: query, max_results: 3 // 限制只抓 3 筆 }) }); const searchData = await searchRes.json(); // ⭐ 關鍵修正：嚴格限制每筆內容的長度，避免 Token 爆炸 let contextStr = searchData.results.map(r => { let shortContent = r.content.substring(0, 500); // 每篇只取前 500 字 return `來源: ${r.title}\n內容: ${shortContent}`; }).join("\n\n"); conversationHistory.push({ role: "tool", tool_call_id: toolCall.id, name: "web_search", content: contextStr }); updateStatus(`🧠 正在總結...`); // --- 第三次請求：回傳總結 --- const finalRes = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.1-8b-instant", // 同樣建議用 8B messages: conversationHistory }) }); const finalData = await finalRes.json(); if (!finalRes.ok) throw new Error(finalData.error?.message); const finalAnswer = finalData.choices[0].message.content; updateChatbox(`🤖 **AI (已聯網):**\n${finalAnswer}\n`); conversationHistory.push({ role: "assistant", content: finalAnswer }); } else { const answer = message.content; updateChatbox(`🤖 **AI:**\n${answer}\n`); conversationHistory.push({ role: "assistant", content: answer }); } } catch (e) { updateChatbox(`\n❌ **[系統錯誤]:** ${e.message}\n`); // 發生錯誤時，把剛才推入歷史紀錄的訊息移除，避免下次發問直接卡死 conversationHistory.pop(); } updateStatus(""); }
