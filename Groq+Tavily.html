// --- 功能二：核心問答邏輯 (修正版) --- let conversationHistory = []; // 用來記憶上下文對話 async function askAI() { const groqKey = document.getElementById('groqKey').value.trim(); const tavilyKey = document.getElementById('tavilyKey').value.trim(); const userInputBox = document.getElementById('userInput'); const userInput = userInputBox.value.trim(); if (!groqKey) return alert("請至少填寫 Groq API Key"); if (!userInput) return alert("請輸入問題"); saveKeys(); // 送出時自動儲存 Key updateChatbox(`\n👤 **你:** ${userInput}\n`); userInputBox.value = ""; updateStatus("Groq 思考中..."); // 將用戶新問題加入歷史紀錄 conversationHistory.push({ role: "user", content: userInput }); try { // 第一步：準備呼叫 Groq 模型的參數 // 這裡非常關鍵，我們必須動態決定要不要傳送 tools 參數 const requestBody = { model: "llama-3.3-70b-versatile", messages: conversationHistory }; // 如果有輸入 Tavily Key，才加上 tools 定義 if (tavilyKey) { requestBody.tools = [{ type: "function", function: { name: "web_search", description: "當用戶詢問近期新聞、即時資訊或模型不知道的事情時，搜尋網際網路", parameters: { type: "object", properties: { query: { type: "string", description: "用來 Google 搜尋的關鍵字" } }, required: ["query"] } } }]; // 可以選擇加上 tool_choice 讓模型自行決定 (auto 也是預設值) requestBody.tool_choice = "auto"; } // 發送給 Groq const response = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify(requestBody) }); // 錯誤捕捉 (捕捉諸如 Rate Limit 或是 Key 無效的錯誤) if (!response.ok) { const errorData = await response.json(); throw new Error(errorData.error?.message || `API 狀態碼 ${response.status}`); } const data = await response.json(); // 安全檢查：確保 choices 有資料 if (!data.choices || data.choices.length === 0) { throw new Error("Groq API 回傳了空的結果 (無 choices 陣列)"); } const message = data.choices[0].message; // 第二步：判斷是否需要連網搜尋 (且確保 message.tool_calls 存在且是陣列) if (message.tool_calls && Array.isArray(message.tool_calls) && message.tool_calls.length > 0) { const toolCall = message.tool_calls[0]; // 確認模型是不是真的叫出了我們定義的工具 if (toolCall.function.name !== "web_search") { console.warn(`模型嘗試呼叫了未定義的工具: ${toolCall.function.name}`); // 若發生這種罕見情況，我們就當作沒這回事，直接跳過搜尋 throw new Error(`模型嘗試呼叫未知工具: ${toolCall.function.name}`); } const query = JSON.parse(toolCall.function.arguments).query; updateStatus(`🔍 啟動網路搜尋... 關鍵字：[${query}]`); conversationHistory.push(message); // 把模型呼叫工具的動作存起來 // 第三步：呼叫 Tavily 搜尋 const searchRes = await fetch("https://api.tavily.com/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ api_key: tavilyKey, query: query, include_answer: false }) }); if (!searchRes.ok) throw new Error("Tavily 搜尋失敗，請檢查 API Key 或是額度"); const searchData = await searchRes.json(); let contextStr = searchData.results.map(r => `標題: ${r.title}\n內容: ${r.content}`).join("\n\n"); updateStatus(`🧠 搜尋完成，Groq 正在整理資料...`); // 將搜尋結果加進對話紀錄中 (身分為 tool) conversationHistory.push({ role: "tool", tool_call_id: toolCall.id, content: contextStr }); // 第四步：拿著搜尋結果，再問一次 Groq const finalRes = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.3-70b-versatile", messages: conversationHistory }) }); if (!finalRes.ok) { const errorData = await finalRes.json(); throw new Error(`(總結階段) ${errorData.error?.message || `API 狀態碼 ${finalRes.status}`}`); } const finalData = await finalRes.json(); if (!finalData.choices || finalData.choices.length === 0) { throw new Error("總結階段 Groq API 回傳了空的結果"); } const finalAnswer = finalData.choices[0].message.content; updateChatbox(`🤖 **AI (已聯網):**\n${finalAnswer}\n`); conversationHistory.push({ role: "assistant", content: finalAnswer }); } else { // 模型認為不需要搜尋，直接回答 const answer = message.content; // 有時候 message.content 可能是 null (雖然在沒有 tool calls 時很少見)，加個防護 if (answer) { updateChatbox(`🤖 **AI:**\n${answer}\n`); conversationHistory.push({ role: "assistant", content: answer }); } else { updateChatbox(`🤖 **AI:** (無文字回應)\n`); } } } catch (e) { updateChatbox(`\n❌ **[系統錯誤]:** 發生問題了！\n原因：${e.message}\n`); // 若發生錯誤，把最後一筆發問移除，避免污染對話上下文 conversationHistory.pop(); } updateStatus(""); // 清空狀態 }
