async function askAI() { const groqKey = document.getElementById('groqKey').value.trim(); const tavilyKey = document.getElementById('tavilyKey').value.trim(); const userInputBox = document.getElementById('userInput'); const userInput = userInputBox.value.trim(); if (!groqKey || !userInput) return alert("請填寫 Groq API Key 與問題"); saveKeys(); updateChatbox(`\n👤 **你:** ${userInput}\n`); userInputBox.value = ""; updateStatus("Groq 思考中..."); conversationHistory.push({ role: "user", content: userInput }); try { const requestBody = { model: "llama-3.1-8b-instant", messages: conversationHistory }; if (tavilyKey) { requestBody.tools = [{ type: "function", function: { name: "web_search", description: "Search the web for current events, news, or real-time info.", parameters: { type: "object", properties: { query: { type: "string", description: "The search query." } }, required: ["query"] } } }]; requestBody.tool_choice = "auto"; } const response = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify(requestBody) }); if (!response.ok) { const errorData = await response.json(); if (response.status === 429) throw new Error("額度暫時上限，請等 1 分鐘。"); throw new Error(errorData.error?.message || `狀態碼 ${response.status}`); } const data = await response.json(); const message = data.choices[0].message; if (message.tool_calls && message.tool_calls.length > 0) { const toolCall = message.tool_calls[0]; const query = JSON.parse(toolCall.function.arguments).query; updateStatus(`🔍 啟動網路搜尋... [${query}]`); conversationHistory.push(message); // ⭐ 關鍵修改一：限制只抓 2 筆資料 const searchRes = await fetch("https://api.tavily.com/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ api_key: tavilyKey, query: query, include_answer: false, max_results: 2 }) }); if (!searchRes.ok) throw new Error("Tavily 搜尋失敗"); const searchData = await searchRes.json(); let contextStr = searchData.results.map(r => `標題: ${r.title}\n內容: ${r.content}`).join("\n\n"); // ⭐ 關鍵修改二：嚴格強制截斷！最多只給 Groq 看前 600 個字 if (contextStr.length > 600) { contextStr = contextStr.substring(0, 600) + "\n...(已強制截斷至 600 字以節省額度)"; } updateStatus(`🧠 Groq 正在總結搜尋結果...`); conversationHistory.push({ role: "tool", tool_call_id: toolCall.id, content: contextStr }); const finalRes = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.1-8b-instant", messages: conversationHistory }) }); if (!finalRes.ok) throw new Error("總結階段發生錯誤"); const finalData = await finalRes.json(); const finalAnswer = finalData.choices[0].message.content; updateChatbox(`🤖 **AI (已聯網):**\n${finalAnswer}\n`); conversationHistory.push({ role: "assistant", content: finalAnswer }); } else { const answer = message.content; updateChatbox(`🤖 **AI:**\n${answer}\n`); conversationHistory.push({ role: "assistant", content: answer }); } } catch (e) { updateChatbox(`\n❌ **[系統錯誤]:** ${e.message}\n`); conversationHistory.pop(); } updateStatus(""); }
