async function askAI() { const groqKey = document.getElementById('groqKey').value.trim(); const tavilyKey = document.getElementById('tavilyKey').value.trim(); const userInputBox = document.getElementById('userInput'); const userInput = userInputBox.value.trim(); if (!groqKey || !userInput) return alert("請輸入 API Key 與問題"); saveKeys(); updateChatbox(`\n👤 **你:** ${userInput}\n`); userInputBox.value = ""; updateStatus("Groq 思考中..."); // 將使用者的問題加入歷史紀錄 conversationHistory.push({ role: "user", content: userInput }); try { // --- 第一步：詢問 Groq 是否需要聯網 --- const response = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.1-8b-instant", // 使用 8B 模型避免 TPM 限制 messages: conversationHistory, tools: [{ type: "function", function: { name: "web_search", description: "當用戶詢問即時新聞、天氣、最新賽事或需要查詢事實時使用此工具。", parameters: { type: "object", properties: { query: { type: "string", description: "搜尋關鍵字" } }, required: ["query"] } } }], tool_choice: "auto" }) }); const data = await response.json(); if (!response.ok) throw new Error(data.error?.message || "Groq 第一階段請求失敗"); const message = data.choices[0].message; // --- 判斷是否觸發搜尋 --- if (message.tool_calls) { const toolCall = message.tool_calls[0]; const query = JSON.parse(toolCall.function.arguments).query; updateStatus(`🔍 聯網搜尋中：${query}...`); // 必須將 Assistant 的 tool_calls 存入歷史 conversationHistory.push(message); // --- 第二步：執行 Tavily 搜尋 --- const searchRes = await fetch("https://api.tavily.com/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ api_key: tavilyKey, query: query, max_results: 3, search_depth: "basic" }) }); if (!searchRes.ok) throw new Error("Tavily API 搜尋失敗，請檢查 Key 或餘額"); const searchData = await searchRes.json(); // ⭐ 極致壓縮：每筆資料只取前 400 字，徹底解決 TPM 爆量問題 let contextStr = searchData.results.map(r => { let cleanContent = (r.content || "").substring(0, 400).replace(/\s+/g, ' '); return `來源: ${r.title}\n內容: ${cleanContent}`; }).join("\n\n"); // 回傳搜尋結果給 AI conversationHistory.push({ role: "tool", tool_call_id: toolCall.id, name: "web_search", content: contextStr }); updateStatus(`🧠 搜尋完成，正在生成回答...`); // --- 第三步：將搜尋結果送回 Groq 進行總結 --- const finalRes = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.1-8b-instant", messages: conversationHistory }) }); const finalData = await finalRes.json(); if (!finalRes.ok) throw new Error(finalData.error?.message || "總結階段失敗"); const finalAnswer = finalData.choices[0].message.content; updateChatbox(`🤖 **AI (已聯網):**\n${finalAnswer}\n`); conversationHistory.push({ role: "assistant", content: finalAnswer }); } else { // 不需要聯網，直接回答 const answer = message.content; updateChatbox(`🤖 **AI:**\n${answer}\n`); conversationHistory.push({ role: "assistant", content: answer }); } } catch (e) { // 錯誤處理：從歷史紀錄中移除最後一筆，避免卡死 updateChatbox(`\n❌ **[系統錯誤]:** ${e.message}\n`); console.error("Debug Info:", e); if (conversationHistory.length > 1) conversationHistory.pop(); } updateStatus(""); }
