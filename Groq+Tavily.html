async function askAI() { const groqKey = document.getElementById('groqKey').value.trim(); const tavilyKey = document.getElementById('tavilyKey').value.trim(); const userInputBox = document.getElementById('userInput'); const userInput = userInputBox.value.trim(); if (!groqKey || !userInput) return alert("請填寫 Groq API Key 與問題"); saveKeys(); updateChatbox(`\n👤 **你:** ${userInput}\n`); userInputBox.value = ""; updateStatus("Groq 思考中..."); conversationHistory.push({ role: "user", content: userInput }); try { // 第一步：問 Groq 是否需要搜尋 const response = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.3-70b-versatile", messages: conversationHistory, tools: [{ type: "function", function: { name: "web_search", description: "Search web for real-time info", parameters: { type: "object", properties: { query: { type: "string" } }, required: ["query"] } } }], tool_choice: "auto" }) }); const data = await response.json(); if (!response.ok) throw new Error(data.error?.message || "Groq 請求失敗"); const message = data.choices[0].message; if (message.tool_calls) { const toolCall = message.tool_calls[0]; const query = JSON.parse(toolCall.function.arguments).query; updateStatus(`🔍 正在搜尋：${query}...`); // 重要：必須把 Assistant 的 tool_calls 存入紀錄 conversationHistory.push(message); // 第二步：呼叫 Tavily // 注意：若遇到 CORS 報錯，請確認你是否在本地 server 執行 const searchRes = await fetch("https://api.tavily.com/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ api_key: tavilyKey, query: query, max_results: 5, search_depth: "basic" }) }); if (!searchRes.ok) { const errText = await searchRes.text(); throw new Error("Tavily 搜尋失敗: " + errText); } const searchData = await searchRes.json(); const contextStr = searchData.results.map(r => `來源: ${r.title}\n內容: ${r.content}`).join("\n\n"); // 重要：存入 tool 角色訊息，必須附帶 tool_call_id conversationHistory.push({ role: "tool", tool_call_id: toolCall.id, name: "web_search", content: contextStr }); updateStatus(`🧠 搜尋完成，正在總結...`); // 第三步：把結果送回 Groq 總結 const finalRes = await fetch("https://api.groq.com/openai/v1/chat/completions", { method: "POST", headers: { "Authorization": `Bearer ${groqKey}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "llama-3.3-70b-versatile", messages: conversationHistory }) }); const finalData = await finalRes.json(); const finalAnswer = finalData.choices[0].message.content; updateChatbox(`🤖 **AI (已聯網):**\n${finalAnswer}\n`); conversationHistory.push({ role: "assistant", content: finalAnswer }); } else { const answer = message.content; updateChatbox(`🤖 **AI:**\n${answer}\n`); conversationHistory.push({ role: "assistant", content: answer }); } } catch (e) { updateChatbox(`\n❌ **[系統錯誤]:** ${e.message}\n`); console.error("詳細錯誤資訊:", e); } updateStatus(""); }
